43 bool ForwardGlobalPlanner::makePlan(const geometry_msgs::PoseStamped& start,

45 {
46  //ROS_WARN_STREAM("Forward global plan goal: " << goal);
47 
48  //three stages: 1 - heading to goal position, 2 - going forward keep orientation, 3 - heading to goal orientation
49 
50  // 1 - heading to goal position
51  // orientation direction
52 
53  double dx = goal.pose.position.x - start.pose.position.x;
54  double dy = goal.pose.position.y - start.pose.position.y;
55 
56  double lenght = sqrt(dx*dx + dy*dy);
57 
58  geometry_msgs::PoseStamped prevState;
59  if (lenght > skip_straight_motion_distance_)
60  {
61  // skip initial pure spinning and initial straight motion
62  //ROS_INFO("1 - heading to goal position pure spinning");
63  double heading_direction = atan2(dy, dx);
64  prevState = reel_path_tools::makePureSpinningSubPlan(start,heading_direction,plan,puresSpinningRadStep_);
65  //ROS_INFO("2 - going forward keep orientation pure straight");
66  prevState = reel_path_tools::makePureStraightSubPlan(prevState, goal.pose.position, lenght, plan);
67  }
68  else
69  {
70  prevState = start;
71  }
72 
73  //ROS_INFO("3 - heading to goal orientation");
74  double goalOrientation = angles::normalize_angle(tf::getYaw(goal.pose.orientation));
75  reel_path_tools::makePureSpinningSubPlan(prevState,goalOrientation,plan,puresSpinningRadStep_);
76 
77  nav_msgs::Path planMsg;
78  planMsg.poses = plan;
79  planMsg.header.stamp = ros::Time::now();
80  planMsg.header.frame_id="/odom";
81  planPub_.publish(planMsg);
82  //ROS_INFO_STREAM("global forward plan: " << planMsg);
83
