129 def load_tmpl(filename):
130  """
131  looks up file within rosdoc ROS package,
132  return its content, may sys.exit on error.
133  Contents are cached with filename as key.
134 
135  :returns: cached file contents
136  """
137  filename = os.path.join(get_templates_dir(), filename)
138  if not os.path.isfile(filename):
139  sys.stderr.write("Cannot locate template file '%s'\n" % (filename))
140  sys.exit(1)
141  with open(filename, 'r') as f:
142  content = f.read()
143  if not content:
144  sys.stderr.write("Template file '%s' is empty\n" % (filename))
145  sys.exit(1)
146  return content
