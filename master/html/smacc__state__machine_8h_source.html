67  template <typename T>

69  {
70  std::lock_guard<std::mutex> lock(m_mutex_);
71  //ROS_WARN("get SM Data lock acquire");

73 
74  if(!globalData_.count(name))

95  }

97  //ROS_WARN("get SM Data lock release");
98  return success;

114  {
115  std::string stateFieldName =demangleSymbol(typeid(StateField).name());
116  std::string behaviorType =demangleSymbol(typeid(BehaviorType).name());
117  ROS_INFO("Mapping state field '%s' to stateBehavior '%s'", stateFieldName.c_str(), behaviorType.c_str());
118  SmaccSubStateBehavior* globalreference;

123  BehaviorType* behavior;

127  this->setGlobalSMData(stateFieldName, globalreference);
