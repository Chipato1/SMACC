75  template <typename T>
76  bool getGlobalSMData(std::string name, T &ret)
77  {
78  std::lock_guard<std::mutex> lock(m_mutex_);
79  //ROS_WARN("get SM Data lock acquire");
80  bool success = false;
81 
82  if (!globalData_.count(name))
83  {
84  //ROS_WARN("get SM Data - data do not exist");
85  success = false;
86  }
87  else
88  {
89  //ROS_WARN("get SM DAta -data exist. accessing");
90  try
91  {
92  auto &v = globalData_[name];
93 
94  //ROS_WARN("get SM DAta -data exist. any cast");
95  ret = boost::any_cast<T>(v.second);
96  success = true;
97  //ROS_WARN("get SM DAta -data exist. success");
98  }
99  catch (boost::bad_any_cast &ex)
100  {
101  ROS_ERROR("bad any cast: %s", ex.what());
102  success = false;
103  }
104  }
105 
106  //ROS_WARN("get SM Data lock release");
107  return success;
108  }

110  template <typename T>

117  globalData_[name] = {

130  template <typename StateField, typename BehaviorType>
131  void mapBehavior()
132  {
133  std::string stateFieldName = demangleSymbol(typeid(StateField).name());
134  std::string behaviorType = demangleSymbol(typeid(BehaviorType).name());
135  ROS_INFO("Mapping state field '%s' to stateBehavior '%s'", stateFieldName.c_str(), behaviorType.c_str());
136  SmaccSubStateBehavior *globalreference;
137  if (!this->getGlobalSMData(stateFieldName, globalreference))
138  {
139  // Using the requires component approach, we force a unique existence
140  // of this component
141  BehaviorType *behavior;
142  this->requiresComponent(behavior);
143  globalreference = dynamic_cast<SmaccSubStateBehavior *>(behavior);
144 
145  this->setGlobalSMData(stateFieldName, globalreference);
146  }
147  }
