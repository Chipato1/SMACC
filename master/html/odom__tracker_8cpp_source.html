149 bool OdomTracker::updateBackward(const nav_msgs::Odometry &odom)
150 {
151  // we initially accept any message if the queue is empty
153  geometry_msgs::PoseStamped base_pose;
154 
155  base_pose.pose = odom.pose.pose;
156  base_pose.header = odom.header;
157  baseTrajectory_.header = odom.header;
158 
159  bool acceptBackward = false;
160  bool pullingerror = false;
161  if (baseTrajectory_.poses.empty())
162  {
163  acceptBackward = false;
164  }
165  else
166  {
167  const geometry_msgs::Point &prevPoint = baseTrajectory_.poses.back().pose.position;
168  const geometry_msgs::Point &currePoint = base_pose.pose.position;
169  double lastpointdist = p2pDistance(prevPoint, currePoint);
170 
171  acceptBackward = !baseTrajectory_.poses.empty() && lastpointdist < minPointDistanceBackwardThresh_;
172 
173  pullingerror = lastpointdist > 2 * minPointDistanceBackwardThresh_;
174  }
175 
176  //ROS_INFO("Backwards, last distance: %lf < %lf accept: %d", dist, minPointDistanceBackwardThresh_, acceptBackward);
177  if (acceptBackward)
178  {
179  baseTrajectory_.poses.pop_back();
180  }
181  else if (pullingerror)
182  {
183  ROS_INFO_THROTTLE(2, "Incorrect backwards motion.");
184  }
185  else
186  {
188  }
189 
190  return acceptBackward;
191 }

198 {

244 void OdomTracker::processOdometryMessage(const nav_msgs::Odometry &odom)
245 {
