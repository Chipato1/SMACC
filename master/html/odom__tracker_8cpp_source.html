158 bool OdomTracker::updateBackward(const nav_msgs::Odometry& odom)
159 {
160  // we initially accept any message if the queue is empty
162  geometry_msgs::PoseStamped base_pose;
163 
164  base_pose.pose = odom.pose.pose;
165  base_pose.header = odom.header;
166  baseTrajectory_.header = odom.header;
167 
168  bool acceptBackward = false;
169  bool pullingerror = false;
170  if(baseTrajectory_.poses.empty())
171  {
172  acceptBackward=false;
173  }
174  else
175  {
176  const geometry_msgs::Point& prevPoint = baseTrajectory_.poses.back().pose.position;
177  const geometry_msgs::Point& currePoint = base_pose.pose.position;
178  double lastpointdist = p2pDistance(prevPoint, currePoint);
179 
180  acceptBackward = !baseTrajectory_.poses.empty()
181  && lastpointdist < minPointDistanceBackwardThresh_;
182 
183  pullingerror = lastpointdist > 2 * minPointDistanceBackwardThresh_;
184  }
185 
186  //ROS_INFO("Backwards, last distance: %lf < %lf accept: %d", dist, minPointDistanceBackwardThresh_, acceptBackward);
187  if (acceptBackward)
188  {
189  baseTrajectory_.poses.pop_back();
190  }
191  else if (pullingerror) {
192  ROS_INFO_THROTTLE(2,"Incorrect backwards motion.");
193  }
194  else
195  {
197  }
198 
199  return acceptBackward;
