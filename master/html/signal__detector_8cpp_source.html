149 void SignalDetector::pollOnce()
150 {
151  for (ISmaccActionClient *smaccActionClient : openRequests_)
152  {
153  // check feedback messages
154  if (smaccActionClient->hasFeedback())
155  {
156  notifyFeedback(smaccActionClient);
157  }
158 
159  // check result
160  auto state = smaccActionClient->getState();
161  if (state.isDone())
162  {
163  finalizeRequest(smaccActionClient);
164  }
165  }
166 }
