138 void clamp(geometry_msgs::Twist& cmd_vel, double max_linear_x_speed_, double max_angular_z_speed_)
139 {
140  if(max_angular_z_speed_==0 || max_linear_x_speed_ ==0)
141  return;
142 
143  if(cmd_vel.angular.z == 0)
144  {
145  cmd_vel.linear.x = max_linear_x_speed_;
146  }
147  else
148  {
149  double kurvature = cmd_vel.linear.x/cmd_vel.angular.z;
150 
151  double linearAuthority= fabs(cmd_vel.linear.x / max_linear_x_speed_) ;
152  double angularAuthority = fabs(cmd_vel.angular.z /max_angular_z_speed_);
153  if( linearAuthority < angularAuthority)
154  {
155  // lets go to maximum linear speed
156  cmd_vel.linear.x = max_linear_x_speed_;
157  cmd_vel.angular.z = kurvature/max_linear_x_speed_;
158  ROS_WARN_STREAM("k="<< kurvature <<"lets go to maximum linear capacity: "<< cmd_vel);
159  }
160  else
161  {
162  // lets go with maximum angular speed
163  cmd_vel.angular.x = max_angular_z_speed_;
164  cmd_vel.linear.x = kurvature*max_angular_z_speed_;
165  ROS_WARN_STREAM("lets go to maximum angular capacity: "<< cmd_vel);
166  }
167  }

176 bool ForwardLocalPlanner::computeVelocityCommands(geometry_msgs::Twist& cmd_vel)
177 {
178  goalReached_=false;
179  //ROS_DEBUG("LOCAL PLANNER LOOP");
180 
181  tf::Stamped<tf::Pose> tfpose = optionalRobotPose(costmapRos_);
182 
183  tf::Quaternion q = tfpose.getRotation();
184 
185  bool ok = false;
186  while (!ok)
187  {
188  // iterate the point from the current position and ahead until reaching a new goal point in the path
189  for (; !ok && currentPoseIndex_ < plan_.size(); currentPoseIndex_++) {
190  auto& pose = plan_[currentPoseIndex_];
191  const geometry_msgs::Point& p = pose.pose.position;
192  tf::Quaternion q;
193  tf::quaternionMsgToTF(pose.pose.orientation, q);
194 
195  // take error from the current position to the path point
196  double dx = p.x - tfpose.getOrigin().x();
197  double dy = p.y - tfpose.getOrigin().y();
198  double dist = sqrt(dx * dx + dy * dy);
199 
200  double pangle = tf::getYaw(q);
201  double angle = tf::getYaw(tfpose.getRotation());
202  double angular_error = angles::shortest_angular_distance(pangle, angle);
203 
204  if (dist >= carrot_distance_|| angular_error >0.1)
205  {
206  // the target pose is enough different to be defined as a target
207  ok = true;
208  //ROS_INFO("forward: %lf", 100.0 * currentPoseIndex_ / (double)plan_.size());
209  }
210  }
211 
212  if (currentPoseIndex_ >= plan_.size()) {
213  // even the latest point is quite similar, then take the last since it is the final goal
214  cmd_vel.linear.x = 0;
215  cmd_vel.angular.z = 0;
216  //ROS_INFO("End Local planner");
217  ok = true;
218  currentPoseIndex_ = plan_.size() -1;
219  //return true;
220  }
221  }
222 
223  //ROS_INFO("pose control algorithm");
224 
225  const geometry_msgs::PoseStamped& goalpose = plan_[currentPoseIndex_];
226  const geometry_msgs::Point& goalposition = goalpose.pose.position;
227 
228  tf::Quaternion goalQ;
229  tf::quaternionMsgToTF(goalpose.pose.orientation, goalQ);
230  //ROS_INFO_STREAM("Plan goal quaternion at "<< goalpose.pose.orientation);
231 
232  //goal orientation (global frame)
233  double betta = tf::getYaw(goalpose.pose.orientation)+ betta_offset_;
234 
235  double dx = goalposition.x - tfpose.getOrigin().x();
236  double dy = goalposition.y - tfpose.getOrigin().y();
237 
238  //distance error to the targetpoint
239  double rho_error = sqrt(dx * dx + dy * dy);
240 
241  //current angle
242  double theta = tf::getYaw(q);
243  double alpha = atan2(dy, dx);
244  alpha = alpha + alpha_offset_;
245 
246  double alpha_error = angles::shortest_angular_distance(alpha, theta);
247  double betta_error = angles::shortest_angular_distance(betta, theta);
248 
249  double vetta;// = k_rho_ * rho_error;
250  double gamma ;//= k_alpha_ * alpha_error + k_betta_ * betta_error;
251 
252  if (rho_error > xy_goal_tolerance_)
253  {
254  vetta = k_rho_ * rho_error;
255  gamma = k_alpha_ * alpha_error;
256  }
257  else if (fabs(betta_error) >= 0.01)
258  {
259  vetta = 0;
260  gamma = k_betta_*betta_error;
261  }
262  else
263  {
264  vetta = 0;
265  gamma = 0;
266  goalReached_=true;
267  }
268 
269  // linear speed clamp
270  if (vetta > max_linear_x_speed_)
271  {
272  vetta = max_linear_x_speed_;
273  }
274  else if(vetta < -max_linear_x_speed_)
275  {
276  vetta = -max_linear_x_speed_;
277  }
278 
279  // angular speed clamp
280  if(gamma > max_angular_z_speed_)
281  {
282  gamma = max_angular_z_speed_;
283  }
284  else if(gamma < -max_angular_z_speed_)
285  {
286  gamma = - max_angular_z_speed_;
287  }
288 
289  cmd_vel.linear.x = vetta;
290  cmd_vel.angular.z = gamma;
291 
292  //clamp(cmd_vel, max_linear_x_speed_, max_angular_z_speed_);
293 
294  //ROS_INFO_STREAM("Local planner: "<< cmd_vel);
295 
296  publishGoalMarker(goalposition.x, goalposition.y, betta);
297 
298  ROS_DEBUG_STREAM("Forward local planner," << std::endl
299  << " theta: " << theta << std::endl
300  << " betta: " << betta << std::endl
301  << " err_x: " << dx << std::endl
302  << " err_y:" << dy << std::endl
303  << " rho_error:" << rho_error << std::endl
304  << " alpha_error:" << alpha_error << std::endl
305  << " betta_error:" << betta_error << std::endl
306  << " vetta:" << vetta << std::endl
307  << " gamma:" << gamma << std::endl
308  << "xy_goal_tolerance:" << xy_goal_tolerance_);
309 
310  //if(cmd_vel.linear.x==0 && cmd_vel.angular.z == 0 )
311  //{
312  //}
313 
314  return true;
