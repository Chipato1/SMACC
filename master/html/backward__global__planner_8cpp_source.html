120 bool BackwardGlobalPlanner::createPureSpiningAndStragihtLineBackwardPath(const geometry_msgs::PoseStamped& start,
121 const geometry_msgs::PoseStamped& goal, std::vector<geometry_msgs::PoseStamped>& plan)
122 {
123  //tf::Stamped<tf::Pose> tfpose;
124  //ROS_WARN_NAMED("Backwards", "getting robot pose referencing costmap: %ld", (long)costmap_ros_);
125 
126  auto q = start.pose.orientation;
127 
128  double dx = start.pose.position.x - goal.pose.position.x ;
129  double dy = start.pose.position.y - goal.pose.position.y ;
130 
131  double lenght = sqrt(dx*dx + dy*dy);
132 
133  geometry_msgs::PoseStamped prevState;
134  if (lenght > skip_straight_motion_distance_)
135  {
136  // skip initial pure spinning and initial straight motion
137  //ROS_INFO("1 - heading to goal position pure spinning");
138  double heading_direction = atan2(dy, dx) ;
139  double startyaw = tf::getYaw(q);
140  double offset = angles::shortest_angular_distance(startyaw, heading_direction);
141  heading_direction = startyaw+offset;
142 
143  prevState = smacc_move_base_client::makePureSpinningSubPlan(start, heading_direction, plan, puresSpinningRadStep_);
144  //ROS_INFO("2 - going forward keep orientation pure straight");
145 
146  prevState = smacc_move_base_client::makePureStraightSubPlan(prevState, goal.pose.position, lenght, plan);
147  }
148  else
149  {
150  prevState = start;
151  }
152 }

159 bool BackwardGlobalPlanner::createDefaultBackwardPath(const geometry_msgs::PoseStamped& start,
160 const geometry_msgs::PoseStamped& goal, std::vector<geometry_msgs::PoseStamped>& plan)
161 {
162  auto q = start.pose.orientation;
163 
164  geometry_msgs::PoseStamped pose;
165  pose = start;
166 
167  plan.push_back(pose);
168 
169  //ROS_WARN_NAMED("Backwards", "Iterating in last forward cord path");
170  int i=lastForwardPathMsg_.poses.size();
171  double mindist =std::numeric_limits<double>::max();
172  int mindistindex = -1;
173 
174  geometry_msgs::Pose goalProjected;
175 
176  for (auto& p : lastForwardPathMsg_.poses | boost::adaptors::reversed)
177  {
178  pose = p;
179  pose.header.frame_id = costmap_ros_->getGlobalFrameID();
180  pose.header.stamp = ros::Time::now();
181 
182  double dx = pose.pose.position.x - goal.pose.position.x;
183  double dy = pose.pose.position.y - goal.pose.position.y;
184 
185  double dist = sqrt(dx*dx + dy*dy);
186  if(dist <= mindist)
187  {
188  mindistindex = i;
189  mindist = dist;
190  goalProjected = pose.pose;
191  }
192 
193  i--;
194  }
195 
196  if(mindistindex != -1)
197  {
198  for (int i = lastForwardPathMsg_.poses.size() -1 ; i>=mindistindex ;i--)
199  {
200  auto& pose = lastForwardPathMsg_.poses[i];
201  plan.push_back(pose);
202  }
203  }
204  else
205  {
206  ROS_WARN_STREAM("Creating the backwards plan, it is not found any close trajectory point. Last forward path plan message size: " << lastForwardPathMsg_.poses.size());
207  }
